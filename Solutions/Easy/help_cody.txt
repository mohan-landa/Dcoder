#include <stdio.h>

//Compiler version gcc  6.3.0

int main()
{
  int n;
  scanf("%d",&n);
  int a[n];
  for(int i=0;i<n;i++){
    scanf("%d",&a[i]);
  }
  for(int i=0;i<n;i++){
    for(int j=i;j<n;j++){
      if(a[i]<a[j]){
        int t = a[j];
        a[j] = a[i];
        a[i] = t;
      }
    }
  }
  for(int i=0;i<n;i++){
    printf("%d ",a[i]);
  }
  return 0;
}
